private void authenticateAndCache(String jwt) {
  if (!jwtProvider.validateAccessToken(jwt)) return;

  Claims claims = jwtProvider.getClaimsFromAccessToken(jwt);
  String userId = claims.getSubject();
  String email = claims.get("email", String.class);

  CustomUserDetails userDetails = new CustomUserDetails(
      UUIDv7.fromString(userId), email, "", false);
  authTokenCache.put(jwt, userDetails, jwtProvider.getAccessTokenExpirationMs());
  setAuthentication(userDetails);
}

AsyncMessagePublishingDecorator - Virtual Thread + Semaphore 패턴 
public class AsyncMessagePublishingDecorator {
  private final ExecutorService executor;
  private final Semaphore semaphore;

  public void executeAsync(Runnable publishTask) {
    executor.execute(() -> {
      try {
        semaphore.acquire();
        try { publishTask.run(); }
        finally { semaphore.release(); }
      } catch (InterruptedException e) {
        Thread.currentThread().interrupt();
      }
    });
  }
}
